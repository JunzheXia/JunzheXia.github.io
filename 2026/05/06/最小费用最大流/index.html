最小费用最大流

无明确上下界，默认下界为0，上界为cap。在保证最大流的前提下，使得总费用最小。这里使用SPFA，反向边是负权边，如果要用Dijkstra要添加势能。

算法思想

使用 SPFA 算法不断寻找单位费用总和最小的增广路（即最短路），然后在每条最短路上进行增广，直到无法继续增广为止。

时间复杂度

  • 最坏情况：$O(F \cdot E \cdot V)$，其中 $F$ 为最大流量
  • 使用 Dijkstra 优化后：$O(F \cdot E \log V)$

完整代码

```cpp

include

define int long long

using namespace std;
using i64 = long long;
using u64 = unsigned long long;

const int N = 5e3 + 10;
const int M = 5e4 + 10;
const int INF = 1e18;

struct Edge {
int to, rev; // 目标节点，反向边索引
int cap, cost; // 剩余容量，单位费用
};

int n, m, s, t;
vector g[N];
int dis[N], cur[N]; // dis: 最短距离, cur: 当前弧优化
bool vis[N];
int mxflow, mncost; // 最大流量，最小费用

void addedge(int u, int v, int cap, int cost) {
// 正向边：容量 cap，费用 cost
g[u].push_back({v, (int)g[v].size(), cap, cost});
// 反向边：容量 0，费用 -cost
g[v].push_back({u, (int)g[u].size() - 1, 0, -cost});
}

bool spfa() {
for (int i = s; i <= t; i++) {
dis[i] = INF;
vis[i] = 0;
cur[i] = 0;
}

queue<int> q;
q.push(s);
dis[s] = 0;
vis[s] = 1;

while (!q.empty()) {
    int u = q.front();
    q.pop();
    vis[u] = 0;

    for (auto &e : g[u]) {
        int v = e.to;
        if (e.cap > 0 && dis[v] > dis[u] + e.cost) {
            dis[v] = dis[u] + e.cost;
            if (!vis[v]) {
                vis[v] = 1;
                q.push(v);
            }
        }
    }
}

return dis[t] != INF;  // 是否能到达汇点

}

int dfs(int u, int flow) {
if (u == t) {
return flow;
}
vis[u] = 1;
int used = 0;

for (int &i = cur[u]; i < (int)g[u].size(); i++) {
    auto &e = g[u][i];
    int v = e.to;

    if (!vis[v] && e.cap > 0 && dis[v] == dis[u] + e.cost) {
        int f = dfs(v, min(flow - used, e.cap));
        if (f) {
            mncost += f * e.cost;  // 累加费用
            e.cap -= f;            // 正向边容量减少
            g[v][e.rev].cap += f;  // 反向边容量增加
            used += f;
            if (used == flow) {
                break;
            }
        }
    }
}
vis[u] = 0;
return used;

}

void dinic() {
while (spfa()) {
memset(vis, 0, sizeof(vis));
int f = dfs(s, INF);
mxflow += f;
}
}

signed main() {
ios::sync_with_stdio(false);
cin.tie(nullptr);

cin >> n >> m >> s >> t;

for (int i = 1; i <= m; i++) {
    int u, v, cap, cost;
    cin >> u >> v >> cap >> cost;
    addedge(u, v, cap, cost);
}

dinic();

cout << mxflow << ' ' << mncost << '\n';

return 0;

}
